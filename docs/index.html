CI/CD Workflow


name: PR Deploy
on: [pull_request]
jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - run: docker buildx build ...
      - run: kubectl create ns rapidrar-pr-${{ pr.number }}
      - run: python test_k8s.py  # <-- Automated Validation
